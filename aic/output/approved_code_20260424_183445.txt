// файл: app/src/main/java/com/printer/cyberdial/HomeScreenViewModel.kt
package com.printer.cyberdial

import androidx.lifecycle.LiveData
import androidx.lifecycle.MutableLiveData
import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.launch
import kotlinx.coroutines.withContext

class HomeScreenViewModel(
    private val repository: AppsRepository
) : ViewModel() {
    
    private val _appsList = MutableLiveData<List<Any>>()
    val appsList: LiveData<List<Any>> = _appsList
    
    private val _widgetsData = MutableLiveData<WidgetsData>()
    val widgetsData: LiveData<WidgetsData> = _widgetsData
    
    init {
        loadData()
    }
    
    private fun loadData() {
        viewModelScope.launch {
            val apps = withContext(Dispatchers.IO) {
                repository.getAppsList()
            }
            
            val widgets = WidgetsData(
                weatherData = "Солнечно 23°",
                musicData = "Трек не выбран",
                calendarData = "Событий нет",
                mamulyaTimer = "03:15"
            )
            
            val combinedList = mutableListOf<Any>()
            combinedList.add(widgets)
            combinedList.addAll(apps)
            
            _widgetsData.postValue(widgets)
            _appsList.postValue(combinedList)
        }
    }
    
    fun updateMamulyaTimer(time: String) {
        val currentWidgets = _widgetsData.value ?: return
        val updatedWidgets = currentWidgets.copy(mamulyaTimer = time)
        _widgetsData.value = updatedWidgets
        
        val currentList = _appsList.value?.toMutableList() ?: return
        if (currentList.isNotEmpty() && currentList[0] is WidgetsData) {
            currentList[0] = updatedWidgets
            _appsList.value = currentList
        }
    }
}

// файл: app/src/main/java/com/printer/cyberdial/MainScreenAdapter.kt
package com.printer.cyberdial

import android.view.LayoutInflater
import android.view.ViewGroup
import androidx.recyclerview.widget.RecyclerView
import com.printer.cyberdial.databinding.ItemAppIconBinding
import com.printer.cyberdial.databinding.WidgetsHeaderBinding

class MainScreenAdapter(
    private val onAppClick: (AppModel, Int) -> Unit,
    private val onWidgetClick: (Int) -> Unit
) : RecyclerView.Adapter<RecyclerView.ViewHolder>() {
    
    companion object {
        const val TYPE_WIDGETS = 0
        const val TYPE_APP = 1
    }
    
    private var items: List<Any> = emptyList()
    
    fun submitList(list: List<Any>) {
        items = list
        notifyDataSetChanged()
    }
    
    override fun getItemViewType(position: Int): Int {
        return if (position == 0 && items.isNotEmpty() && items[0] is WidgetsData) {
            TYPE_WIDGETS
        } else {
            TYPE_APP
        }
    }
    
    override fun onCreateViewHolder(parent: ViewGroup, viewType: Int): RecyclerView.ViewHolder {
        return if (viewType == TYPE_WIDGETS) {
            val binding = WidgetsHeaderBinding.inflate(LayoutInflater.from(parent.context), parent, false)
            WidgetsViewHolder(binding, onWidgetClick)
        } else {
            val binding = ItemAppIconBinding.inflate(LayoutInflater.from(parent.context), parent, false)
            AppViewHolder(binding, onAppClick)
        }
    }
    
    override fun onBindViewHolder(holder: RecyclerView.ViewHolder, position: Int) {
        when (holder) {
            is WidgetsViewHolder -> {
                val widgets = items[position] as WidgetsData
                holder.bind(widgets)
            }
            is AppViewHolder -> {
                val app = items[position] as AppModel
                val adapterPosition = if (getItemViewType(position) == TYPE_WIDGETS) position - 1 else position
                holder.bind(app, adapterPosition)
            }
        }
    }
    
    override fun getItemCount(): Int = items.size
}

class AppViewHolder(
    private val binding: ItemAppIconBinding,
    private val onAppClick: (AppModel, Int) -> Unit
) : RecyclerView.ViewHolder(binding.root) {
    
    fun bind(app: AppModel, position: Int) {
        binding.appIcon.setImageResource(app.iconRes)
        binding.appName.text = app.name
        
        if (!app.isAvailable) {
            binding.appIcon.alpha = 0.5f
            binding.ivLock.visibility = android.view.View.VISIBLE
        } else {
            binding.appIcon.alpha = 1.0f
            binding.ivLock.visibility = android.view.View.GONE
        }
        
        if (app.isFolder) {
            binding.appIcon.alpha = 1.0f
            binding.ivLock.visibility = android.view.View.GONE
        }
        
        binding.root.setOnClickListener {
            animateClick()
            onAppClick(app, position)
        }
    }
    
    private fun animateClick() {
        binding.root.animate()
            .scaleX(0.95f)
            .scaleY(0.95f)
            .setDuration(100)
            .withEndAction {
                binding.root.animate()
                    .scaleX(1.0f)
                    .scaleY(1.0f)
                    .setDuration(100)
                    .start()
            }
            .start()
    }
}

// файл: app/src/main/java/com/printer/cyberdial/WidgetsViewHolder.kt
package com.printer.cyberdial

import android.os.Handler
import android.os.Looper
import android.view.View
import androidx.recyclerview.widget.RecyclerView
import com.printer.cyberdial.databinding.WidgetsHeaderBinding

class WidgetsViewHolder(
    private val binding: WidgetsHeaderBinding,
    private val onWidgetClick: (Int) -> Unit
) : RecyclerView.ViewHolder(binding.root) {
    
    private var timerHandler: Handler? = null
    private var timerRunnable: Runnable? = null
    private var remainingSeconds = 60
    
    fun bind(widgetsData: WidgetsData) {
        binding.tvWeather.text = widgetsData.weatherData
        binding.tvMusic.text = widgetsData.musicData
        binding.tvCalendar.text = widgetsData.calendarData
        binding.tvMamulya.text = widgetsData.mamulyaTimer
        
        binding.cardWeather.setOnClickListener { onWidgetClick(0) }
        binding.cardMusic.setOnClickListener { onWidgetClick(1) }
        binding.cardCalendar.setOnClickListener { onWidgetClick(2) }
        binding.cardMamulya.setOnClickListener { onWidgetClick(3) }
        
        startMamulyaTimer()
    }
    
    private fun startMamulyaTimer() {
        stopMamulyaTimer()
        
        timerHandler = Handler(Looper.getMainLooper())
        timerRunnable = object : Runnable {
            override fun run() {
                if (remainingSeconds > 0) {
                    remainingSeconds--
                    val minutes = remainingSeconds / 60
                    val seconds = remainingSeconds % 60
                    val timeString = String.format("%02d:%02d", minutes, seconds)
                    binding.tvMamulya.text = "Мамуля скоро напишет... $timeString"
                    timerHandler?.postDelayed(this, 1000)
                } else {
                    binding.tvMamulya.text = "Мамуля скоро напишет... 00:00"
                }
            }
        }
        timerHandler?.postDelayed(timerRunnable!!, 1000)
    }
    
    private fun stopMamulyaTimer() {
        timerRunnable?.let { timerHandler?.removeCallbacks(it) }
        timerHandler = null
        timerRunnable = null
    }
    
    fun onDestroy() {
        stopMamulyaTimer()
    }
}

package com.printer.cyberdial

import android.os.Bundle
import android.view.LayoutInflater
import android.view.View
import android.view.ViewGroup
import android.widget.ImageView
import android.widget.TextView
import android.widget.Toast
import androidx.recyclerview.widget.GridLayoutManager
import androidx.recyclerview.widget.RecyclerView
import com.google.android.material.bottomsheet.BottomSheetDialogFragment
import com.printer.cyberdial.databinding.DialogFolderBinding
import com.printer.cyberdial.databinding.ItemAppIconBinding

class FolderDialogFragment : BottomSheetDialogFragment() {
    
    private var _binding: DialogFolderBinding? = null
    private val binding get() = _binding!!
    
    private lateinit var folder: AppModel
    
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        folder = arguments?.getSerializable("folder") as AppModel
    }
    
    override fun onCreateView(
        inflater: LayoutInflater,
        container: ViewGroup?,
        savedInstanceState: Bundle?
    ): View {
        _binding = DialogFolderBinding.inflate(inflater, container, false)
        return binding.root
    }
    
    override fun onViewCreated(view: View, savedInstanceState: Bundle?) {
        super.onViewCreated(view, savedInstanceState)
        
        binding.tvFolderTitle.text = folder.name
        
        val recyclerView = binding.recyclerViewFolder
        recyclerView.layoutManager = GridLayoutManager(requireContext(), 3)
        
        val folderApps = folder.appsInFolder ?: emptyList()
        val adapter = FolderAdapter(folderApps) { app ->
            handleAppClick(app)
        }
        
        recyclerView.adapter = adapter
    }
    
    private fun handleAppClick(app: AppModel) {
        if (!app.isAvailable) {
            Toast.makeText(requireContext(), "Приложение заблокировано", Toast.LENGTH_SHORT).show()
        } else if (app.packageName != null) {
            try {
                val intent = requireContext().packageManager.getLaunchIntentForPackage(app.packageName)
                if (intent != null) {
                    startActivity(intent)
                    dismiss()
                } else {
                    Toast.makeText(requireContext(), "Приложение не установлено", Toast.LENGTH_SHORT).show()
                }
            } catch (e: Exception) {
                Toast.makeText(requireContext(), "Ошибка открытия приложения", Toast.LENGTH_SHORT).show()
            }
        } else {
            Toast.makeText(requireContext(), "Открытие ${app.name}", Toast.LENGTH_SHORT).show()
        }
    }
    
    override fun onDestroyView() {
        super.onDestroyView()
        _binding = null
    }
    
    companion object {
        fun newInstance(folder: AppModel): FolderDialogFragment {
            val fragment = FolderDialogFragment()
            val args = Bundle()
            args.putSerializable("folder", folder)
            fragment.arguments = args
            return fragment
        }
    }
}

class FolderAdapter(
    private val apps: List<AppModel>,
    private val onItemClick: (AppModel) -> Unit
) : RecyclerView.Adapter<FolderAdapter.FolderViewHolder>() {
    
    override fun onCreateViewHolder(parent: ViewGroup, viewType: Int): FolderViewHolder {

    }
    
    override fun onBindViewHolder(holder: FolderViewHolder, position: Int) {
        holder.bind(apps[position])
    }
    
    override fun getItemCount(): Int = apps.size
    
    class FolderViewHolder(

        fun bind(app: AppModel) {

package com.printer.cyberdial

import android.os.Bundle
import android.view.LayoutInflater
import android.view.View
import android.view.ViewGroup
import android.widget.Toast
import androidx.fragment.app.Fragment
import androidx.lifecycle.ViewModelProvider
import androidx.recyclerview.widget.GridLayoutManager
import com.printer.cyberdial.databinding.FragmentHomeScreenBinding

class HomeScreenFragment : Fragment() {
    
    private var _binding: FragmentHomeScreenBinding? = null
    private val binding get() = _binding!!
    
    private lateinit var viewModel: HomeScreenViewModel
    private lateinit var adapter: MainScreenAdapter
    private lateinit var widgetsViewHolder: WidgetsViewHolder
    
    override fun onCreateView(
        inflater: LayoutInflater,
        container: ViewGroup?,
        savedInstanceState: Bundle?
    ): View {
        _binding = FragmentHomeScreenBinding.inflate(inflater, container, false)
        return binding.root
    }
    
    override fun onViewCreated(view: View, savedInstanceState: Bundle?) {
        super.onViewCreated(view, savedInstanceState)
        
        val repository = AppsRepository(requireContext())
        viewModel = ViewModelProvider(this, object : ViewModelProvider.Factory {
            override fun <T : androidx.lifecycle.ViewModel> create(modelClass: Class<T>): T {
                return HomeScreenViewModel(repository) as T
            }
        }).get(HomeScreenViewModel::class.java)
        
        setupRecyclerView()
        setupDock()
        observeData()
    }
    
    private fun setupRecyclerView() {
        adapter = MainScreenAdapter(
            onAppClick = { app, position ->
                handleAppClick(app)
            },
            onWidgetClick = { widgetType ->
                when (widgetType) {
                    0 -> Toast.makeText(requireContext(), "Прогноз погоды", Toast.LENGTH_SHORT).show()
                    1 -> Toast.makeText(requireContext(), "Музыкальный плеер", Toast.LENGTH_SHORT).show()
                    2 -> Toast.makeText(requireContext(), "Календарь", Toast.LENGTH_SHORT).show()
                    3 -> Toast.makeText(requireContext(), "Сообщение маме", Toast.LENGTH_SHORT).show()
                }
            }
        )
        
        val layoutManager = GridLayoutManager(requireContext(), 5)
        layoutManager.spanSizeLookup = object : GridLayoutManager.SpanSizeLookup() {
            override fun getSpanSize(position: Int): Int {
                return if (position == 0 && adapter.getItemViewType(position) == MainScreenAdapter.TYPE_WIDGETS) {
                    5
                } else {
                    1
                }
            }
        }
        
        binding.recyclerViewApps.layoutManager = layoutManager
        binding.recyclerViewApps.adapter = adapter
    }
    
    private fun setupDock() {
        binding.ivPhone.setOnClickListener {
            Toast.makeText(requireContext(), "Телефон", Toast.LENGTH_SHORT).show()
        }
        binding.ivMessages.setOnClickListener {
            Toast.makeText(requireContext(), "Сообщения", Toast.LENGTH_SHORT).show()
        }
        binding.ivBrowser.setOnClickListener {
            Toast.makeText(requireContext(), "Браузер", Toast.LENGTH_SHORT).show()
        }
        binding.ivCamera.setOnClickListener {
            Toast.makeText(requireContext(), "Камера", Toast.LENGTH_SHORT).show()
        }
    }
    
    private fun observeData() {
        viewModel.appsList.observe(viewLifecycleOwner) { list ->
            adapter.submitList(list)
        }
    }
    
    private fun handleAppClick(app: AppModel) {
        if (app.isFolder) {
            val dialog = FolderDialogFragment.newInstance(app)
            dialog.show(parentFragmentManager, "folder_dialog")
        } else if (!app.isAvailable) {
            Toast.makeText(requireContext(), "Приложение заблокировано", Toast.LENGTH_SHORT).show()
        } else if (app.packageName != null) {
            try {
                val intent = requireContext().packageManager.getLaunchIntentForPackage(app.packageName)
                if (intent != null) {
                    startActivity(intent)
                } else {
                    Toast.makeText(requireContext(), "Приложение не установлено", Toast.LENGTH_SHORT).show()
                }
            } catch (e: Exception) {
                Toast.makeText(requireContext(), "Ошибка открытия приложения", Toast.LENGTH_SHORT).show()
            }
        } else {
            Toast.makeText(requireContext(), "Открытие ${app.name}", Toast.LENGTH_SHORT).show()
        }
    }
    
    override fun onDestroyView() {
